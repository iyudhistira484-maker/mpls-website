FIRESTORE SECURITY RULES
========================

Paste the following rules into Firebase Console > Firestore Database > Rules tab

---

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Helper function to check if user is admin
    function isAdmin() {
      return get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }

    // Helper function to check if user is authenticated
    function isAuthenticated() {
      return request.auth != null;
    }

    // Allow read for authenticated users on all documents
    match /{document=**} {
      allow read: if isAuthenticated();
    }

    // Users collection
    match /users/{userId} {
      // Users can read their own data
      allow read: if request.auth.uid == userId;
      
      // Admins can read all user data
      allow read: if isAdmin();
      
      // Users can update their own data
      allow update: if request.auth.uid == userId;
      
      // Admins can update any user data
      allow update: if isAdmin();
      
      // Anyone can create a user account (registration)
      allow create: if isAuthenticated();
      
      // Admins can delete users
      allow delete: if isAdmin();
    }

    // OTP codes collection - public for verification
    match /otp_codes/{email} {
      allow read: if true;
      allow write: if true;
      allow delete: if true;
    }

    // Schedules collection - admin write, all authenticated read
    match /schedules/{scheduleId} {
      allow read: if isAuthenticated();
      allow create: if isAdmin();
      allow update: if isAdmin();
      allow delete: if isAdmin();
    }

    // Materials collection - admin write, all authenticated read
    match /materials/{materialId} {
      allow read: if isAuthenticated();
      allow create: if isAdmin();
      allow update: if isAdmin();
      allow delete: if isAdmin();
    }

    // Submissions collection - students can write their own, admins can read all
    match /submissions/{submissionId} {
      allow read: if isAuthenticated();
      allow create: if isAuthenticated();
      allow update: if isAuthenticated() || isAdmin();
      allow delete: if isAdmin();
    }

    // Attendance collection - students can write their own, admins can read all
    match /attendance/{attendanceId} {
      allow read: if isAuthenticated();
      allow create: if isAuthenticated();
      allow update: if isAdmin();
      allow delete: if isAdmin();
    }

    // Quiz collection - admin write, all authenticated read
    match /quiz/{quizId} {
      allow read: if isAuthenticated();
      allow create: if isAdmin();
      allow update: if isAdmin();
      allow delete: if isAdmin();
    }

    // Announcements collection - admin write, all authenticated read
    match /announcements/{announcementId} {
      allow read: if isAuthenticated();
      allow create: if isAdmin();
      allow update: if isAdmin();
      allow delete: if isAdmin();
    }

    // Gallery collection - admin write, all authenticated read
    match /gallery/{galleryId} {
      allow read: if isAuthenticated();
      allow create: if isAdmin();
      allow update: if isAdmin();
      allow delete: if isAdmin();
    }
  }
}

STORAGE RULES
=============

Paste the following rules into Firebase Console > Storage > Rules tab

---

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    
    // Helper function to check if user is admin
    function isAdmin() {
      return get(/databases/(default)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }

    // Gallery uploads - admin only
    match /gallery/{allPaths=**} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && isAdmin();
    }

    // Submissions uploads - students can upload their own
    match /submissions/{userId}/{allPaths=**} {
      allow read: if request.auth.uid == userId || isAdmin();
      allow write: if request.auth.uid == userId;
    }

    // Default deny all other paths
    match /{allPaths=**} {
      allow read, write: if false;
    }
  }
}

---

NOTES:
- Replace {database} with your actual database ID (usually "default")
- Test these rules thoroughly before deploying to production
- For production, consider more restrictive rules
- Monitor rule violations in Firebase Console > Firestore > Rules > Metrics
